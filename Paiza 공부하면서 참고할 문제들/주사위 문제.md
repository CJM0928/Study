## 주사위 풀기위한 기본적인 틀

```
package test.test;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;

class Dice {

	int top;
	int bottom;
	int west;
	int east;
	int north;
	int south;
	
	Dice(int top, int bottom, int west, int east, int north, int south){
		this.top = top;
		this.bottom = bottom;
		this.west = west;
		this.east = east;
		this.north = north;
		this.south = south;
	}
	
	public void rollEast() {
		int t = top; 				// top은 west로
		int b = bottom; 			// bottom은 east가 되어야함
		int w = west;				// west는 top으로
		int e = east;				// east는 bottom으로
		
		top = w;
		bottom = e;
		west = t;
		east = b;
	}
	
	public void rollWest() {
		int t = top;				// top은 east로
		int b = bottom;				// bottom은 west로
		int w = west;				// west는 bottom으로
		int e = east;				// east는 top으로
		
		top = e;
		bottom = w;
		west = b;
		east = t;
	}
	
	public void rollNorth() {
		int t = top;				// top은 south로
		int b = bottom;				// bottom은 north으로
		int n = north;				// north는 bottom으로
		int s = south;				// south는 top으로
		
		top = s;
		bottom = n;
		north = t;
		south = b;
	}
	
	public void rollSouth() {
		int t = top;				// top은 north로
		int b = bottom;				// bottom은 south로
		int n = north;				// north는 top으로
		int s = south;				// south는 bottom으로
		
		top = n;
		bottom = s;
		north = b;
		south = t;
	}
}

public class Test2 {
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st = new StringTokenizer(br.readLine());
		
		//T B U D L R
		int t = Integer.parseInt(st.nextToken());
		int b = Integer.parseInt(st.nextToken());
		int u = Integer.parseInt(st.nextToken());
		int d = Integer.parseInt(st.nextToken());
		int l = Integer.parseInt(st.nextToken());
		int r = Integer.parseInt(st.nextToken());
		
		st = new StringTokenizer(br.readLine());
		// 입력: T B U D L R
		// 생성자: (top, bottom, west, east, north, south)
		Dice dice = new Dice(t, b, l, r, d, u);
		int n = Integer.parseInt(st.nextToken());
		
		int moveCount = 0;
		
		for(int i=0; i<n; i++) {
			st = new StringTokenizer(br.readLine());
			int target = Integer.parseInt(st.nextToken());
			if(target == dice.top) {
				continue;
			} else if(target == dice.east) {
				dice.rollWest();
				moveCount++;
			} else if(target == dice.west) {
				dice.rollEast();
				moveCount++;
			} else if(target == dice.north) {
				dice.rollSouth();
				moveCount++;
			} else if(target == dice.south) {
				dice.rollNorth();
				moveCount++;
			} else {
				dice.rollNorth();
				dice.rollNorth();
				moveCount += 2;
			}
		}
		
		System.out.println(moveCount);
	}
}
```
